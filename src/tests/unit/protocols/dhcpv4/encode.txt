#
#  ----  IPv4 prefix with split format  ----

#
#  ----  IPv4 prefix with bits format (in struct)  ----
#
#  Classless-Static-Route (121 = 0x79) struct member Subnet uses prefix=bits.
#  Encoded as: prefix_len(1) + ceil(prefix/8) bytes of address + router(4)
#

#
#  /0 prefix (default route) - no address bytes
#
encode-pair Classless-Static-Route = { Destination-Descriptor = { Subnet = 0.0.0.0/0, Router-IP-Address = 10.0.0.1 } }
match 79 05 00 0a 00 00 01

decode-pair -
match Classless-Static-Route = { Destination-Descriptor = { Subnet = 0.0.0.0/0, Router-IP-Address = 10.0.0.1 } }

#
#  /32 host route - full 4 address bytes
#
encode-pair Classless-Static-Route = { Destination-Descriptor = { Subnet = 10.0.0.1/32, Router-IP-Address = 10.0.0.254 } }

decode-pair -
match Classless-Static-Route = { Destination-Descriptor = { Subnet = 10.0.0.1/32, Router-IP-Address = 10.0.0.254 } }
