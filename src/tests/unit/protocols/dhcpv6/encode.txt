#
#  ----  Standalone ipv6addr (S46-Border-Relay)  ----
#
#  Option 90 (0x005a) - simple ipv6addr without struct context.
#

encode-pair S46-Border-Relay = 2001:db8::1
match 00 5a 00 10 20 01 0d b8 00 00 00 00 00 00 00 00 00 00 00 01

decode-pair -
match S46-Border-Relay = 2001:db8::1

encode-pair S46-Border-Relay = fe80::1
match 00 5a 00 10 fe 80 00 00 00 00 00 00 00 00 00 00 00 00 00 01

decode-pair -
match S46-Border-Relay = fe80::1

#
#  ----  Standalone ipv6prefix variable length (S46-Default-Mapping-Rule)  ----
#
#  Option 91 (0x005b) - variable-length encoding: prefix_len(1) + ceil(prefix/8) bytes.
#  Unlike ipv6prefix inside structs (which always writes 17 bytes), standalone
#  ipv6prefix uses variable-length encoding.
#

#
#  /0 prefix - just the prefix length byte, no address bytes
#
encode-pair S46-Default-Mapping-Rule = ::/0
match 00 5b 00 01 00

decode-pair -
match S46-Default-Mapping-Rule = ::/0

#
#  /32 prefix - 4 address bytes
#
encode-pair S46-Default-Mapping-Rule = 2001:db8::/32
match 00 5b 00 05 20 20 01 0d b8

decode-pair -
match S46-Default-Mapping-Rule = 2001:db8::/32

#
#  /48 prefix - 6 address bytes
#
encode-pair S46-Default-Mapping-Rule = 2001:db8:abcd::/48
match 00 5b 00 07 30 20 01 0d b8 ab cd

decode-pair -
match S46-Default-Mapping-Rule = 2001:db8:abcd::/48

#
#  /128 prefix - full 16 address bytes
#
encode-pair S46-Default-Mapping-Rule = ::1/128
match 00 5b 00 11 80 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 01

decode-pair -
match S46-Default-Mapping-Rule = ::1/128

#
#  ----  Partial DNS label (Client-FQDN with multi-label FQDN)  ----
#
#  RFC 4704 Client-FQDN Domain-Name has partial_dns_label flag.
#  The trailing null byte is stripped for both single-label and
#  multi-label names (RFC 4704 says partial name omits trailing zero).
#
#  Existing tests (rfc4704.txt) cover single-label "tapioca01".
#  This tests multi-label FQDN to verify trailing null is also stripped.
#

encode-pair Client-FQDN.No-Server-Update = no, Client-FQDN.Server-Override = no, Client-FQDN.Server-Update = yes, Client-FQDN.Domain-Name = "host.example.com"
match 00 27 00 12 01 04 68 6f 73 74 07 65 78 61 6d 70 6c 65 03 63 6f 6d

decode-pair -
match Client-FQDN = { Reserved = 0, No-Server-Update = no, Server-Override = no, Server-Update = yes, Domain-Name = "host.example.com" }

#
#  ----  Struct with ipv4prefix + ipv6prefix (S46-Rule)  ----
#
#  Option 89 (0x0059) - struct encoding with mixed prefix types.
#  In structs, ipv4prefix is fixed 5 bytes and ipv6prefix is fixed 17 bytes.
#

encode-pair S46-Rule.Flags = 1, S46-Rule.IPv4-Prefix = 192.168.0.0/24, S46-Rule.IPv6-Prefix = 2001:db8::/32
match 00 59 00 17 01 18 c0 a8 00 00 20 20 01 0d b8 00 00 00 00 00 00 00 00 00 00 00 00

decode-pair -
match S46-Rule = { Flags = 1, IPv4-Prefix = 192.168.0.0/24, IPv6-Prefix = 2001:db8::/32 }
